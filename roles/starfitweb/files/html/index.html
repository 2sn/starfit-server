#!/usr/bin/env python import header header.http() import glob from os import path import jinja2 as j2 try: from starfit import __version__ as starfit_version except ImportError: starfit_version = 'unknown' # Find DB files db_files = glob.glob('/var/www/html/data/db/*.stardb.*') # Dict of labels of DB files db_labels = {} # Look for labels file db_labels_file = '/var/www/html/data/db/labels' if path.isfile(db_labels_file): with open(db_labels_file, 'r') as f: # Read each line, and put the filename and label in the dict for line in f.readlines(): cols = line.split('=') db_labels[cols[0].strip()] = cols[1].strip() # If there are files that do not have a label, generate one for them based on the filename for fpath in db_files: fname = path.basename(fpath) if fname not in db_labels: db_labels[fname] = fname.split('.stardb.')[0] # Find Star files star_files = glob.glob('/var/www/html/data/stars/*.dat') # Dict of labels of star files star_labels = {} # Look for labels file star_labels_file = '/var/www/html/data/stars/labels' if path.isfile(star_labels_file): with open(star_labels_file, 'r') as f: # Read each line, and put the filename and label in the dict for line in f.readlines(): cols = line.split('=') star_labels[cols[0].strip()] = cols[1].strip() # If there are files that do not have a label, generate one for them based on the filename for fpath in star_files: fname = path.basename(fpath) if fname not in star_labels: star_labels[fname] = fname[:-4] # Load jinja template jinja_env = j2.Environment(loader=j2.FileSystemLoader("templates")) template = jinja_env.get_template("home.html") # Render using DB list print(template.render( dblist=db_labels, starfit_version=starfit_version, db_listing_size=str(min(30, len(db_labels))), db_select=next(iter(db_labels)), starlist=star_labels, star_select=next(iter(star_labels)) ))
